✅ T003 — Create Admin Route Guard + Dynamic Admin Tab Router

Create two new components.

1. Create
src/shared/components/RequireAdmin.tsx

Implementation:

import { useAuth } from '@/shared/hooks/use-auth';
import { Navigate } from 'react-router-dom';

export function RequireAdmin({ children }: { children: React.ReactNode }) {
  const { user, isLoading } = useAuth();

  if (isLoading) return null;

  if (user?.role !== 'admin') {
    return <Navigate to="/" replace />;
  }

  return <>{children}</>;
}
2. Create
src/admin/pages/AdminTabPage.tsx

Implementation:

import { useParams, Navigate } from 'react-router-dom';
import { ADMIN_TAB_COMPONENTS, isAdminTab } from '@/admin/AdminPanel';

export default function AdminTabPage() {
  const { tab } = useParams<{ tab: string }>();

  if (!tab || !isAdminTab(tab)) {
    return <Navigate to="/" replace />;
  }

  return <>{ADMIN_TAB_COMPONENTS[tab]}</>;
}
